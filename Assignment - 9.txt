2. Write a menu driven program with the following options:
   a. Addition
   b. Subtraction
   c. Multiplication
   d. Division
   e. Exit

Solution:-

#include<stdio.h>
int main()
{
int num,num1,num2;
float num3,num4,num5;

printf("Choose any One Options = ");
scanf("%d",&num);
printf("\n");
switch(num)
{
default :
    printf("Sorry This Option Doesn't Exist.");
    break;
case 1:
    printf("you have chosen the Addition\nEnter Two Numbers\n");
    scanf("%d%d",&num1,&num2);
    printf("Value is %d",num1+num2);
    break;
case 2:
    printf("you have chosen the Subtraction\nEnter Two Numbers\n");
    scanf("%d%d",&num1,&num2);
    printf("Value is %d",num1-num2);
    break;
case 3:
    printf("you have chosen the Multiplication\nEnter Two Numbers\n");
    scanf("%d%d",&num1,&num2);
    printf("Value is %d",num1*num2);
    break;
case 4:
    printf("you have chosen the Division\nEnter Two Numbers\n");
    scanf("%d%d",&num3,&num4);
    num5=num3/num4;
    printf("Value is %.2f",num5);
    break;
case 5:
    printf("you have chosen the Exit\nThank You");
    break;
}
printf("\n");
return 0;
}

6. Program to check whether a year is a leap year or not. Using switch
statement

Solution:- 

#include<stdio.h>
int main()
{
int year;
printf("Enter Year = ");
scanf("%d",&year);
if(year%400==0)
    year=1;
else if(year%4==0)
    year=1;
else
    year=0;

switch(year)
{
case 1:
    printf("Leap Year");
    break;
case 0:
    printf("Not Leap Year");
    break;
}
printf("\n");
return 0;
}


7. Program to take the value from the user as input electricity unit charges
   and calculate total electricity bill according to the given condition.
   Using the switch statement.
   For the first 50 units Rs. 0.50/unit
   For the next 100 units Rs. 0.75/unit
   For the next 100 units Rs. 1.20/unit
   For units above 250 Rs. 1.50/unit
   An additional surcharge of 20% is added to the bill.

Solution:-

#include<stdio.h>
int main()
{
int x;
float unit;
printf("Enter Meter Reading Unit = ");
scanf("%d",&x);
unit=x;
switch(x)
{
case 1 ... 50:
    unit=0.50*unit;
    unit=unit/5+unit;
    printf("\nTotal Electricity Bill is %.2f\n",unit);
    break;
case 51 ... 150:
    unit=0.75*unit;
    unit=unit/5+unit;
    printf("\nTotal Electricity Bill is %.2f\n",unit);
    break;
case 151 ... 250:
    unit=1.20*unit;
    unit=unit/5+unit;
    printf("\nTotal Electricity Bill is %.2f\n",unit);
    break;
case 251 ... 10000:
    unit=1.50*unit;
    unit=unit/5+unit;

    break;

}
